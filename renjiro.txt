npcHandler:setCallback(CALLBACK_ONTRADEREQUEST, function(cid)
    local player = Player(cid)
    if not player then
        return false
    end

    local renjiroRep = player:getStorageValue(NpcStorageReputation.konoha.renjiro)


    if renjiroRep < 0 then
        player:setStorageValue(NpcStorageReputation.konoha.renjiro,0)
        player:save()
    end

    -- limpa a shop antes de montar novamente
    npcHandler.shopItems = {}

    for _, item in ipairs(sellableItems) do
        shopModule:addSellableItem(item[1], item[2], item[3], item[4], item[5], item[6])
    end

    for _, item in ipairs(buyableItemsByReputation) do
        if renjiroRep >= item[1] then
            shopModule:addBuyableItem(item[2], item[3], item[4], item[5], item[6])
        end
    end

    return true
end)

local validWords = {
    ["pt"] = {"ajudar", "hospital", "ninjas", "aceita", "sim", "coisa", "sementes"},
    ["en"] = {"help", "hospital", "ninjas", "accept", "yes", "thing", "seeds"}
}

local function dialogOfNpc(cid, player, currentLang, npcText, topic, reputationStorageValue, questStorageValue, releasedFocus, effect)
    npcHandler:say(messages[currentLang][npcText], cid)
    npcHandler.topic[cid] = topic

    if reputationStorageValue ~= nil then
        player:setStorageValue(NpcStorageReputation.konoha.renjiro, reputationStorageValue)
        player:save()   
    end


    if questStorageValue ~= nil then
        player:setStorageValue(NpcStorageReputation.konoha.renjiroRepQuest, questStorageValue)
        player:save()      
    end

    
    if releasedFocus == true then
        npcHandler:releaseFocus(cid)      
    end

    if effect ~= nil then
        if effect == 483 then
			local pos = player:getPosition()
			local effectPos = Position(pos.x + 1, pos.y + 1, pos.z)
            effectPos:sendMagicEffect(effect)
            return true
        end
        player:getPosition():sendMagicEffect(effect)
    end
end


function creatureSayCallback(cid, type, msg)
    if not npcHandler:isFocused(cid) then
        return false
    end

    local msgLower = string.lower(msg)
    local player = Player(cid)
    local vocationId = player:getVocation():getId()
    local currentLang = npcHandler.languages[cid] or "pt" -- default language pt
    npcHandler:addFocus(cid)

    local isValid = false
    for _, word in ipairs(validWords[currentLang]) do
        if msgcontains(msgLower, word) then
            isValid = true
            break
        end
    end
    if not isValid then
        return true
    end

    local konohaRep = player:getStorageValue(PlayerStorageReputation.konoha)
    local renjiroRep = player:getStorageValue(NpcStorageReputation.konoha.renjiro)
    local repQuest = player:getStorageValue(NpcStorageReputation.konoha.renjiroRepQuest)

    if konohaRep < 0 then
        konohaRep = 0
        player:setStorageValue(PlayerStorageReputation.konoha, 0)
    end

    if repQuest < 0 then
        repQuest = 0
        player:setStorageValue(NpcStorageReputation.konoha.renjiroRepQuest, 0)
        player:save()
    end

    if renjiroRep < 0 then
        player:setStorageValue(NpcStorageReputation.konoha.renjiro,0)
        player:save()
    end

    -- Comandos de ajuda
    if msgcontains(msgLower, "ajudar") or msgcontains(msgLower, "help") then

        if repQuest == 1 and player:getItemCount(2416) >= 10 and player:getItemCount(2417) >= 10 and player:getItemCount(2418) >= 10 then
            player:removeItem(2416, 10)
            player:removeItem(2417, 10)
            player:removeItem(2418, 10)
            player:addItem(9725, 10) -- Health Potion
            player:addItem(9728, 10) -- Mana Potion           
            dialogOfNpc(cid, player, currentLang, "m1completed", 1, 1, 2, true, 483)
            return true            
        elseif repQuest == 1 and player:getItemCount(2418) < 10 then
            dialogOfNpc(cid, player, currentLang, "m1noitem", 1, nil, nil, false, nil)
            return true
        end

        if repQuest == 3 and player:getItemCount(2419) >= 10 then
            player:removeItem(2419, 10)
            player:addItem(9726, 10) -- Strong Health Potion
            player:addItem(9729, 10) -- Strong Mana Potion          
            dialogOfNpc(cid, player, currentLang, "m2completed", 1, 2, 4, true, 483)
            return true
        elseif repQuest == 3 and player:getItemCount(2419) < 10 then
            dialogOfNpc(cid, player, currentLang, "m2noitem", 1, nil, nil, false, nil)
            return true
        end

        dialogOfNpc(cid, player, currentLang, "greeting", 0, nil, nil, false, nil)
        return true
    end

    ----------------------------------------------------------------------------------------------------------------------------------------------------------
    -- Reputation Mission 1: Collect Flowers
    if npcHandler.topic[cid] == 0 and msgcontains(msgLower, "hospital") and repQuest == 0 then
        if renjiroRep == 0 and konohaRep == 0 then
            dialogOfNpc(cid, player, currentLang, "talkHospital", 1, nil, nil, false, nil)
        return true
        else
            dialogOfNpc(cid, player, currentLang, "noRep", 1, nil, nil, false, nil)
            return true
        end
    end

    if npcHandler.topic[cid] == 1 and msgcontains(msgLower, "ninjas") and repQuest == 0 then
        dialogOfNpc(cid, player, currentLang, "help1", 2, nil, nil, false, nil)
        return true
    end

    if npcHandler.topic[cid] == 2 and (msgcontains(msgLower, "aceita") or msgcontains(msgLower, "accept") or msgcontains(msgLower, "sim") or msgcontains(msgLower, "yes"))  and repQuest == 0 then
        dialogOfNpc(cid, player, currentLang, "m1", 3, nil, 1, true, 526)
        return true        
    end
    

    ----------------------------------------------------------------------------------------------------------------------------------------------------------
    -- Reputation Mission 2: Collect the seeds
    if npcHandler.topic[cid] == 0 and msgcontains(msgLower, "hospital") and repQuest == 2 then
        if renjiroRep == 1 and konohaRep >= 1 then
            dialogOfNpc(cid, player, currentLang, "start2", 1, nil, nil, false, nil)
        return true
        else
            dialogOfNpc(cid, player, currentLang, "noRep", 1, nil, nil, false, nil)
            return true
        end        
    end

    if npcHandler.topic[cid] == 1 and (msgcontains(msgLower, "coisa") or msgcontains(msgLower, "thing")) and repQuest == 2 then
        dialogOfNpc(cid, player, currentLang, "help2", 2, nil, nil, false, nil)
        return true
    end

    if npcHandler.topic[cid] == 2 and (msgcontains(msgLower, "sementes") or msgcontains(msgLower, "seeds") or msgcontains(msgLower, "accept") or msgcontains(msgLower, "sim")) and repQuest == 2 then
        dialogOfNpc(cid, player, currentLang, "m2", 0, 3, nil, 3, true, 526)
        return true        
    end

    return true
end

npcHandler:setCallback(CALLBACK_MESSAGE_DEFAULT, creatureSayCallback)
