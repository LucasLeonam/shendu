local keywordHandler = KeywordHandler:new()
local npcHandler = NpcHandler:new(keywordHandler)
NpcSystem.parseParameters(npcHandler)

function onCreatureAppear(cid)              npcHandler:onCreatureAppear(cid)            end
function onCreatureDisappear(cid)           npcHandler:onCreatureDisappear(cid)         end
function onCreatureSay(cid, type, msg)      npcHandler:onCreatureSay(cid, type, msg)    end
function onThink()                          npcHandler:onThink()                        end

local shopModule = ShopModule:new()
npcHandler:addModule(shopModule)

keywordHandler:addKeyword({'stuff'}, StdModule.say, {
    npcHandler = npcHandler,
    text = 'Just ask me for a {trade} to see my offers.'
})
keywordHandler:addAliasKeyword({'wares'})
keywordHandler:addAliasKeyword({'offer'})

npcHandler:setCallback(CALLBACK_ONTRADEREQUEST, function(cid)
    local player = Player(cid)
    if not player then
        return false
    end

    local shizuneRep = player:getStorageValue(NpcStorageReputation.konoha.shizune)


    if shizuneRep < 0 then
        player:setStorageValue(NpcStorageReputation.konoha.shizune,0)
        player:save()
    end

    -- limpa a shop antes de montar novamente
    npcHandler.shopItems = {}


    -- Reputação 0 ou maior
    shopModule:addBuyableItem({'health potion', 'small potion'}, 9725, 80, nil, 'Health Potion', shizuneRep)

    -- Reputação 1 ou maior
    if shizuneRep >= 1 then
        shopModule:addBuyableItem({'mana potion', 'small mana potion'}, 9728, 100, nil, 'Chakra Potion', shizuneRep)
    end

    -- Reputação 2 ou maior
    if shizuneRep >= 2 then
        shopModule:addBuyableItem({'strong health potion', 'medium potion'}, 9726, 250, nil, 'Strong Health Potion', shizuneRep)
        shopModule:addBuyableItem({'strong mana potion', 'medium potion'}, 9729, 300, nil, 'Strong Chakra Potion', shizuneRep)
    end

    -- Reputação 3 ou maior
    if shizuneRep >= 3 then
        shopModule:addBuyableItem({'great health potion', 'great potion'}, 9727, 400, nil, 'Great Chakra Potion', shizuneRep)
        shopModule:addBuyableItem({'great mana potion', 'great mana potion'}, 9730, 450, nil, 'Great Chakra Potion', shizuneRep)
    end

    -- Reputação 4 ou maior
    if shizuneRep >= 4 then
        shopModule:addBuyableItem({'health pill'}, 10464, 550, nil, 'Health Pill', shizuneRep)
    end

    -- Reputação 5 ou maior
    if shizuneRep >= 5 then
        shopModule:addBuyableItem({'chakra pill'}, 10463, 600, nil, 'Chakra Pill', shizuneRep)
    end

    if shizuneRep >= 6 then
        shopModule:addBuyableItem({'fish'}, 2333, 900, nil, 'Fish', shizuneRep)
    end


    return true
end)

local validWords = {
    ["pt"] = {"ajudar", "hospital", "ninjas", "aceita", "sim", "coisa", "sementes", "eficazes", "demanda","especificas", "vida", "importante", "pílulas", "último", "Akimichi"},

local function dialogOfNpc(cid, player, currentLang, npcText, topic, reputationStorageValue, questStorageValue, releasedFocus, effect)
    npcHandler:say(messages[currentLang][npcText], cid)
    npcHandler.topic[cid] = topic

    if reputationStorageValue ~= nil then
        player:setStorageValue(NpcStorageReputation.konoha.shizune, reputationStorageValue)
        player:save()
    end

    if questStorageValue ~= nil then
        player:setStorageValue(NpcStorageReputation.konoha.shizuneRepQuest, questStorageValue)

        local questlineStorages = {
            NpcStorageReputation.konoha.shizuneRepQuestline1,
            NpcStorageReputation.konoha.shizuneRepQuestline2,
            NpcStorageReputation.konoha.shizuneRepQuestline3,
            NpcStorageReputation.konoha.shizuneRepQuestline4,
            NpcStorageReputation.konoha.shizuneRepQuestline5,
            NpcStorageReputation.konoha.shizuneRepQuestline6,
        }

        local questlineIndex = math.ceil(questStorageValue / 2)
        local questlineValue = questStorageValue % 2

        if questlineValue == 0 then
            questlineValue = 2
        end

        local questlineStorage = questlineStorages[questlineIndex]

        if questlineStorage ~= nil then
            player:setStorageValue(questlineStorage, questlineValue)
        end

        player:save()
    end

    if releasedFocus == true then
        npcHandler:releaseFocus(cid)
    end

    if effect ~= nil then
        if effect == 483 then
            local pos = player:getPosition()
            local effectPos = Position(pos.x + 1, pos.y + 1, pos.z)
            effectPos:sendMagicEffect(effect)
            return true
        end

        player:getPosition():sendMagicEffect(effect)
    end
end


function creatureSayCallback(cid, type, msg)
    if not npcHandler:isFocused(cid) then
        return false
    end

    local player = Player(cid)
    local vocationId = player:getVocation():getId()

    npcHandler:addFocus(cid)

    local isValid = false
    for _, word in ipairs(validWords[currentLang]) do
        if msgcontains(msgLower, word) then
            isValid = true
            break
        end

        return true

    local konohaRep = player:getStorageValue(PlayerStorageReputation.konoha)
    local shizuneRep = player:getStorageValue(NpcStorageReputation.konoha.shizune)

    if konohaRep < 0 then
        konohaRep = 0
        player:setStorageValue(PlayerStorageReputation.konoha, 0)
    end

    if repQuest < 0 then
        repQuest = 0
        player:setStorageValue(NpcStorageReputation.konoha.shizuneRepQuest, 0)
        player:save()
    end

    if shizuneRep < 0 then
        shizuneRep = 0
        player:setStorageValue(NpcStorageReputation.konoha.shizune,0)

            player:removeItem(2418, 10)
            player:addItem(9725, 10) -- Health Potion
            player:addItem(9728, 10) -- Mana Potion
